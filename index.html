    <script>

        const MODEL_NAME = "gemini-2.5-flash-preview-09-2025";

        

        // Settings Elements

        const settingsBtn = document.getElementById('settingsBtn');

        const settingsOverlay = document.getElementById('settingsOverlay');

        const apiKeyInput = document.getElementById('apiKeyInput');

        const saveKeyBtn = document.getElementById('saveKeyBtn');

        const closeSettingsBtn = document.getElementById('closeSettingsBtn');


        // Logic Elements

        const audioInput = document.getElementById('audioInput');

        const transcribeBtn = document.getElementById('transcribeBtn');

        const fileNameDisplay = document.getElementById('fileNameDisplay');

        const fileInfo = document.getElementById('fileInfo');

        const resultSection = document.getElementById('resultSection');

        const summaryContent = document.getElementById('summaryContent');

        const transcriptContainer = document.getElementById('transcriptContainer');

        const statusIndicator = document.getElementById('statusIndicator');

        const toast = document.getElementById('toast');

        const toastText = document.getElementById('toastText');


        let selectedFile = null;

        let generatedData = { summary: "", transcript: "" };


        function showToast(msg) {

            toastText.textContent = msg;

            toast.classList.replace('opacity-0', 'opacity-100');

            setTimeout(() => toast.classList.replace('opacity-100', 'opacity-0'), 3000);

        }


        // API Key Management

        function getSavedKey() { return localStorage.getItem('gemini_api_key') || ""; }

        

        settingsBtn.onclick = () => {

            apiKeyInput.value = getSavedKey();

            settingsOverlay.classList.remove('hidden');

        };

        

        closeSettingsBtn.onclick = () => settingsOverlay.classList.add('hidden');

        

        saveKeyBtn.onclick = () => {

            localStorage.setItem('gemini_api_key', apiKeyInput.value.trim());

            showToast('API Key 已儲存');

            settingsOverlay.classList.add('hidden');

        };


        // File Selection

        document.getElementById('dropZone').onclick = () => audioInput.click();

        document.getElementById('dropZone').ondragover = (e) => { e.preventDefault(); e.currentTarget.classList.add('drag-over'); };

        document.getElementById('dropZone').ondragleave = (e) => e.currentTarget.classList.remove('drag-over');

        document.getElementById('dropZone').ondrop = (e) => {

            e.preventDefault(); e.currentTarget.classList.remove('drag-over');

            handleFile(e.dataTransfer.files[0]);

        };

        audioInput.onchange = (e) => handleFile(e.target.files[0]);


        function handleFile(file) {

            if (!file || !file.type.startsWith('audio/')) { showToast('請選擇音訊檔案'); return; }

            selectedFile = file;

            fileNameDisplay.textContent = `${file.name} (${(file.size/1024/1024).toFixed(1)}MB)`;

            fileInfo.classList.remove('hidden');

            transcribeBtn.disabled = false;

        }


        // Transcription API

        async function runTranscription(file, key) {

            const base64 = await new Promise(resolve => {

                const r = new FileReader(); r.readAsDataURL(file);

                r.onload = () => resolve(r.result.split(',')[1]);

            });


            const systemPrompt = "你是 Monika 的 AI 專屬秘書。請針對音訊產出：1. 'summary': 繁體中文摘要(Markdown)。2. 'transcript': 去除贅字的繁體中文完整逐字稿。請回傳 JSON。";

            

            const payload = {

                contents: [{ parts: [

                    { text: "請分析這段錄音並回傳 JSON 格式 (包含 summary 和 transcript 欄位)。" },

                    { inlineData: { mimeType: file.type, data: base64 } }

                ]}],

                systemInstruction: { parts: [{ text: systemPrompt }] },

                generationConfig: { responseMimeType: "application/json" }

            };


            const res = await fetch(`https://generativelanguage.googleapis.com/v1beta/models/${MODEL_NAME}:generateContent?key=${key}`, {

                method: 'POST', body: JSON.stringify(payload)

            });

            if (!res.ok) throw new Error('API Key 無效或請求失敗');

            const data = await res.json();

            return JSON.parse(data.candidates[0].content.parts[0].text);

        }


        transcribeBtn.onclick = async () => {

            const key = getSavedKey();

            if (!key) { showToast('請先在設定中輸入 API Key'); settingsBtn.click(); return; }

            if (!selectedFile) return;


            transcribeBtn.disabled = true;

            statusIndicator.classList.remove('hidden');

            resultSection.classList.remove('hidden');

            summaryContent.innerHTML = `<div class="h-4 w-full loading-shimmer rounded mb-2"></div><div class="h-4 w-2/3 loading-shimmer rounded"></div>`;

            transcriptContainer.textContent = "AI 正在轉錄中...";


            try {

                const data = await runTranscription(selectedFile, key);

                generatedData = data;

                summaryContent.innerHTML = data.summary.replace(/\n/g, '<br>');

                transcriptContainer.textContent = data.transcript;

                showToast('處理完成！');

            } catch (err) {

                showToast('處理失敗：' + err.message);

            } finally {

                statusIndicator.classList.add('hidden');

                transcribeBtn.disabled = false;

            }

        };


        // Export Function

        document.getElementById('exportBtn').onclick = () => {

            if (!generatedData.transcript) return;

            const docContent = `<html xmlns:o='urn:schemas-microsoft-com:office:office' xmlns:w='urn:schemas-microsoft-com:office:word' xmlns='http://www.w3.org/TR/REC-html40'>

                <head><meta charset='utf-8'></head><body>

                <h1 style="color: #0284C7;">Monika 的會議記錄摘要</h1>

                <div style="background: #FFFBEB; padding: 15px;">${generatedData.summary.replace(/\n/g, '<br>')}</div>

                <h2>完整逐字稿</h2><p>${generatedData.transcript}</p>

                </body></html>`;

            const blob = new Blob(['\ufeff', docContent], { type: 'application/msword' });

            const link = document.createElement('a');

            link.href = URL.createObjectURL(blob);

            link.download = `會議紀錄_${new Date().toISOString().slice(0,10)}.doc`;

            link.click();

        };

    </script>
